161void ThreadPool::shutdown()
162{
163 if (!running)
164 {
165 return;
166 }
167
168 running = false;
169
170 Logger::info("[Thread Pool] Shutdown requested...");
171
172 auto start = std::chrono::steady_clock::now();
173 while (!queue.empty())
174 {
175 std::this_thread::sleep_for(std::chrono::milliseconds(5));
176
177 if (std::chrono::steady_clock::now() - start > std::chrono::seconds(1))
178 {
179 Logger::warn("[Thread Pool] Timeout waiting for queue to drain.");
180 break;
181 }
182 }
183
184 queue.shutdown();
185
186 join();
187 Logger::info("[Thread Pool] All workers joined. Shutdown complete.");
188}

311void ThreadPool::threadLoop(const std::string& worker_name)
312{
313 Logger::info("[" + worker_name + "] Started");
314
315 while (true)
316 {
317 std::unique_ptr<IJob> job = queue.pop();
318
319 if (!job)
320 break;
321
322 try
323 {
324 job->execute();
325 }
326 catch (const std::exception& e)
327 {
328 Logger::error("[Thread Pool][" + worker_name + "] Exception: " + e.what());
329 }
330 }
331 Logger::info("[" + worker_name + "] Exiting");
332}
