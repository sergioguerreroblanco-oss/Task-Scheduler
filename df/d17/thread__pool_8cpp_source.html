90void ThreadPool::shutdown()
91{
92 if (!running)
93 {
94 return;
95 }
96
97 running = false;
98
99 Logger::info("[Thread Pool] Shutdown requested...");
100
101 auto start = std::chrono::steady_clock::now();
102 while (!queue.empty())
103 {
104 std::this_thread::sleep_for(std::chrono::milliseconds(5));
105
106 if (std::chrono::steady_clock::now() - start > std::chrono::seconds(1))
107 {
108 Logger::warn("[Thread Pool] Timeout waiting for queue to drain.");
109 break;
110 }
111 }
112
113 queue.shutdown();
114
115 join();
116 Logger::info("[Thread Pool] All threads joined. Shutdown complete.");
117}

183void ThreadPool::threadLoop(const std::string& worker_name)
184{
185 Logger::info("[" + worker_name + "] Started");
186
187 while (true)
188 {
189 std::unique_ptr<IJob> job = queue.pop();
190
191 if (!job)
192 break;
193
194 try
195 {
196 job->execute();
197 }
198 catch (const std::exception& e)
199 {
200 Logger::error("[Thread Pool][" + worker_name + "] Exception: " + e.what());
201 }
202 }
203 Logger::info("[" + worker_name + "] Exiting");
204}
